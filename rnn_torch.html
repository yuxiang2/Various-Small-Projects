class LanguageModelDataLoader(DataLoader):
    def __init__(self, dataset, batch_size, shuffle=True):
        
        self.data, self.target = self.divide(dataset, chunk_size=20)
        self.batch_size = batch_size
        self.shuffle = shuffle

    def divide(self, dataset, chunk_size):
        n_seq = 0
        for article in dataset:
            n_seq += (len(article)-1) // chunk_size
            
        data = np.zeros((n_seq, chunk_size)).astype(int)
        target = np.zeros((n_seq, chunk_size)).astype(int)
        
        idx = 0
        for article in dataset:
            length = (len(article)-1) // chunk_size
            data[idx: idx + length] = article[:length * chunk_size].reshape(length, chunk_size)
            target[idx: idx + length] = article[1:length * chunk_size + 1].reshape(length, chunk_size)
            idx += length
        return (data, target)
        
    def __iter__(self):
        # concatenate your articles and build into batches
        num_data = len(self.data)
        if self.shuffle:
            idx = np.random.permutation(num_data)
        else:
            idx = np.arange(num_data)
        
        for i in range(0, num_data - self.batch_size, self.batch_size):
            data = self.data[idx[i:i+self.batch_size]]
            target = self.target[idx[i:i+self.batch_size]]
            yield (data.T, target.T)
            
        data = self.data[idx[num_data - num_data % self.batch_size:]]
        target = self.target[idx[num_data - num_data % self.batch_size:]]
        yield (data.T, target.T)

class LanguageModel(nn.Module):
    def __init__(self, vocab_size, embed_size = 384, hidden_size = 512, nlayers = 1):
        super(LanguageModel, self).__init__()
        
        self.vocab_size = vocab_size
        self.embed_size = embed_size
        self.hidden_size = hidden_size 
        self.nlayers = nlayers
        self.embedding = nn.Embedding(vocab_size, embed_size)
        self.rnn = nn.LSTM(embed_size, hidden_size, nlayers)
        self.scoring = nn.Linear(hidden_size, vocab_size)

    def forward(self, x, hidden = None):
        batch_size = x.size(1)
        embed = self.embedding(x)
        output, hidden = self.rnn(embed, hidden)
        output_flatten = output.view(-1, self.hidden_size) # (Length x batch) x Hidden
        score = self.scoring(output_flatten)
        return score.view(-1, batch_size, self.vocab_size), hidden

class LanguageModelTrainer:
    def __init__(self, model, loader, max_epochs=1, run_id='exp'):
        """
            Use this class to train your model
        """
        # feel free to add any other parameters here
        self.model = model.cuda()
        self.loader = loader
        self.train_losses = []
        self.val_losses = []
        self.predictions = []
        self.predictions_test = []
        self.generated_logits = []
        self.generated = []
        self.generated_logits_test = []
        self.generated_test = []
        self.epochs = 0
        self.max_epochs = max_epochs
        self.run_id = run_id
        
        # TODO: Define your optimizer and criterion here
        self.optimizer = torch.optim.Adam(model.parameters(), 0.0001)
        self.criterion = torch.nn.CrossEntropyLoss()

    def train(self):
        self.model.train() # set to training mode
        epoch_loss = 0
        num_batches = 0
        for batch_num, (inputs, targets) in enumerate(self.loader):
            epoch_loss += self.train_batch(inputs, targets)
        epoch_loss = epoch_loss / (batch_num + 1)
        self.epochs += 1
        print('[TRAIN]  Epoch [%d/%d]   Loss: %.4f'
                      % (self.epochs, self.max_epochs, epoch_loss))
        self.train_losses.append(epoch_loss)

    def train_batch(self, inputs, targets):
        """ 
            TODO: Define code for training a single batch of inputs
        
        """
        inputs = torch.from_numpy(inputs).long().cuda()
        outputs, _ = self.model(inputs)
        outputs = outputs.view(-1, outputs.size(2))
        targets = torch.from_numpy(targets.reshape(-1)).long().cuda()
        loss = self.criterion(outputs[:], targets)
        self.optimizer.zero_grad()
        loss.backward()
        self.optimizer.step()
        print('loss:{}'.format(loss.item()), end='\r')
        return loss.item()

    
    def test(self):

        self.model.eval() # set to eval mode
        predictions = TestLanguageModel.prediction(fixtures_pred['inp'], self.model) # get predictions

        generated_test = test_generation(fixtures_gen_test, generated_logits_test, vocab)
        self.val_losses.append(nll)
        
        self.generated.append(generated)
        self.generated_test.append(generated_test)
        self.generated_logits.append(generated_logits)
        self.generated_logits_test.append(generated_logits_test)
        
        # generate predictions for test data
        predictions_test = TestLanguageModel.prediction(fixtures_pred_test['inp'], self.model) # get predictions
        self.predictions_test.append(predictions_test)
            
        print('[VAL]  Epoch [%d/%d]   NLL: %.4f'
                      % (self.epochs, self.max_epochs, nll))
        return nll

    def save(self):
